PriceCard component takes in data that fits the following schema set up with Zod

```js
const SubscriptionPriceSchema = z.object({
  monthly: price,
  yearly: price,
  currency: CurrencySymbolSchema,
});

export const SupportServiceSchema = z.object({
  id: z.string(),
  service: z.string(),
  descrip: z.string(),
  priceLevels: z.object({
  monthly: price,
  yearly: price,
  currency: CurrencySymbolSchema,
}),
  level: PlansSchema,
  active: z.boolean(),
  features: z.array(
  z.object({
    copy: z.string(),
    has: z.boolean(),
  }),
);,
});
```
